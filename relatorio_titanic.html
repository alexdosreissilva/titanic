def percentual(fracao, total=712):
    """
        Funo que retorna o percentual correspondente
        de uma amostra em um total de registros.
        Arguments:
            fracao: Quantidade de amostras.
            total: Total de registros.
        Return:
            Um nmero float correspondente ao percentual de uma amostra
            em um total de registros.
    """
    return ceil((fracao * 100) / total)

def pie(explode, labels, sizes):
    """
        Funo que gera grfico de pizza.
        Arguments:
            explode: O espaamento de cada parte do grfico.
            labels: O rotulo de cada parte do grfico.
            sizes: O percentual de cada parte do grfico.
        Return:
            Nenhum.
    """
    explode = explode

    fig1, ax1 = plt.subplots()
    ax1.pie(sizes, explode=explode, labels=labels, autopct='%1.1f%%',
        shadow=True, startangle=90)
    ax1.axis('equal')

    plt.show()

def bar(n_groups, vals_bar_1, vals_bar_2, std, label_bar_1, label_bar_2, 
        color_bar_1, color_bar_2, xlabel, ylabel, title, xticklabels):
    """
        Funo que gera grfico de barras.
        Arguments:
            n_groups: A quantidade de grupos no grfico.
            vals_bar_1: Os valores da primeira barra de cada grupo no grfico.
            vals_bar_2: Os valores da segunda barra de cada grupo no grfico.
            std: O tamanho do trao vertical em cima de cada barra dos grupos no grfico.
            label_bar_1: O rotulo da primeira barra de cada grupo no grfico.
            label_bar_2: O rotulo da segunda barra de cada grupo no grfico.
            color_bar_1: A cor da primeira barra de cada grupo no grfico.
            color_bar_2: A cor da segunda barra de cada grupo no grfico.
            xlabel: O rotulo do eixo x do grfico.
            ylabel: O rotulo do eixo y do grfico.
            title: O titulo do grfico.
            xticklabels: O rotulo de cada grupo no grfico.
        Return:
            Nenhum.
    """
    
    fig, ax = plt.subplots()

    index = np.arange(n_groups)
    bar_width = 0.35

    opacity = 0.4
    error_config = {'ecolor': '0.3'}

    rects1 = ax.bar(index, vals_bar_1, bar_width,
                    alpha=opacity, color=color_bar_1,
                    yerr=std, error_kw=error_config,
                    label=label_bar_1)

    rects2 = ax.bar(index + bar_width, vals_bar_2, bar_width,
                    alpha=opacity, color=color_bar_2,
                    yerr=std, error_kw=error_config,
                    label=label_bar_2)

    ax.set_xlabel(xlabel)
    ax.set_ylabel(ylabel)
    ax.set_title(title)
    ax.set_xticks(index + bar_width / 2)
    ax.set_xticklabels(xticklabels)
    ax.legend()

    fig.tight_layout()
    plt.show()

Conclusões¶

De acordo com a anlise feita dos dados referente ao acidente do titanic, conclui-se que as pessoas do sexo feminino (67.3%) foram mais propensas a sobreviver em relao as pessoas do sexo masculino (32.7%).

As pessoas com idade maior ou igual a idade mediana (51.5%) foram mais privilegiadas a sobreviver em comparao as pessoas com idade menor (48.5%).

A idade mediana dos passageiros do titanic era de 28 anos.

Apenas as pessoas da classe alta (41.6%) tiveram mais vantagem a sobreviver em ralao a classe mdia (28.7%) e classe baixa (29.7%).

As mulheres da classe alta com idade maior ou igual a 28 anos tiveram mais privilegios para sobreviver.
